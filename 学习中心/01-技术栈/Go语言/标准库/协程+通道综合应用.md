# 两个协程交替执行

```go
var wg sync.WaitGroup  
  
func main() {  
    wg.Add(2)  
    c := make(chan int)  
    s := "abcdefghijk"  
    go printS(c, s)  
    c <- 0  
    go printS(c, s)  
    wg.Wait()  
}  
  
func printS(c chan int, s string) {  
    defer wg.Done()  
    for {
       index, ok := <-c  
       if !ok {  
          return  
       }  
       if index >= len(s) {  
          close(c)  
          return  
       }  
       fmt.Println(string(s[index]))  
       c <- index + 1  
    }  
}
```
