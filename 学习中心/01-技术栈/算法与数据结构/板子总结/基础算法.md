# 基础算法

## 1. 前缀和差分

- 一维前缀和

```c++
S[i] = S[i-1] + a[i];
a[l] + ... + a[r] = S[r] - S[l - 1]
```

- 二维前缀和

```c++
s[i][j] = s[i][j-1]+s[i-1][j]-s[i-1][j-1]+a[i][j];
```

- 一维差分

```c++
b[i] = a[i] - a[i-1];
```

- 二维差分

```c++
b[i][j]=a[i][j]-a[i-1][j]-a[i][j-1]+a[i-1][j-1]
```

## 3. 二分

- 整数二分

```c++
while(l<=r) {
    int mid = (l+r)/2;
    if (check(mid)) {
        l=mid+1;
        ans=mid;
    }else r=mid-1;
}
```

- 浮点二分

```c++
while(r-l>=1e-8)
{
    double mid = (l+r)/2;
    if(check(mid)) {
        l=mid;
        ans=mid;
    }
    else r = mid;
}
```

## 3. 字符串

### 1. 字符串hash

```c++
#include<iostream>
 
using namespace std;
 
typedef unsigned long long ULL;
 
const int N=100010,P=131;//注意这里是大p,也可以是13331这个值
 
int n,m;
char str[N];
ULL h[N],p[N];//这里是小p,h[N]代表的是数值，p[N]代表的是存储p的多少次方
 
ULL get(int l,int r)
{
    return h[r]-h[l-1]*p[r-l+1];
}
int main()
{
    scanf("%d%d%s",&n,&m,str+1);//str+1代表数组从下标是1开始存储
 
    p[0]=1;
    for(int i=1;i<=n;i++)
    {
        p[i]=p[i-1]*P;//前面是小p，后面是大p，下面也是一样
        h[i]=h[i-1]*P+str[i];
    }
 
    while(m--)
    {
        int l1,r1,l2,r2;
        scanf("%d%d%d%d",&l1,&r1,&l2,&r2);
 
        if(get(l1,r1)==get(l2,r2)) puts("Yes");
        else puts("No");
    }
    return 0;
}
```
