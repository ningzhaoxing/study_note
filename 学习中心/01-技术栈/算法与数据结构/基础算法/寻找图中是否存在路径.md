# 解题思路
首先将边数组转化为邻接表，方便我们找到起点和递归深度搜索。
然后，我们进行深度优先搜索，递归的出口为遍历的当前节点等于终点节点，意味着我们找到了有效路径。
若遍历完该节点所有的邻接节点，没有找到有效路径，则说明有效路径不通过该节点，返回`false`。

# AC 代码
```go
  
func validPath(n int, edges [][]int, source int, destination int) bool {  
    matrix := make([][]int, n)  
    for _, edge := range edges {  
       x := edge[0]  
       y := edge[1]  
       matrix[x] = append(matrix[x], y)  
       matrix[y] = append(matrix[y], x)  
    }  
    vis := make([]bool, n)  
  
    var dfs func(cur int) bool  
    dfs = func(cur int) bool {  
       if cur == destination {  
          return true  
       }  
       vis[cur] = true  
  
       for i := 0; i < len(matrix[cur]); i++ {  
          if !vis[matrix[cur][i]] && dfs(matrix[cur][i]) {  
             return true  
          }  
       }       return false  
    }  
    return dfs(source)  
}
```
